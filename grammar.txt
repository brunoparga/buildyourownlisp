number   : /-?[0-9]+(\.[0-9]+)?/ ;
symbol   : "list" | "head" | "tail" | "join" | "eval" | "cons" | "length" | '+' | '-' | '*' | '/' | "%" | '^' | "min" | "max" ;
sexpr    : '(' <expr>* ')' ;
qexpr    : '{' <expr>* '}' ;
expr     : <number> | <symbol> | <sexpr> | <qexpr> ;
comment  : ';' /.*/ ;
lye      : /^/ (<expr> | <comment>)* /$/ ;
